# Define functions to assign scores based on given criteria
def assign_basic_knowledge_score(major_category):
    if major_category == "공학계열":
        return 30
    elif major_category == "자연계열":
        return 20
    else:
        return 10

def assign_related_experience_score(it_level):
    if it_level == "개발자 수준":
        return 30
    elif it_level == "보통":
        return 20
    else:
        return 10

def assign_entrepreneurship_intent_score(purpose):
    if "개발역량확보" in purpose or "기초지식습득" in purpose:
        return 20
    elif "창업기회발굴" in purpose:
        return 15
    else:
        return 10

def assign_class_participation_intent_score(introduction):
    if len(introduction) > 100:
        return 20
    elif 50 < len(introduction) <= 100:
        return 15
    else:
        return 10

# Apply scoring functions to each applicant
applicant_data['기본지식(관련학과) 점수'] = applicant_data['Unnamed: 8'].apply(assign_basic_knowledge_score)
applicant_data['관련 경험 점수'] = applicant_data['Unnamed: 10'].apply(assign_related_experience_score)
applicant_data['취창업 의지 점수'] = applicant_data['Unnamed: 12'].apply(assign_entrepreneurship_intent_score)
applicant_data['수업 참여 의지 점수'] = applicant_data['Unnamed: 14'].apply(assign_class_participation_intent_score)

# Calculate total score
applicant_data['총점'] = (applicant_data['기본지식(관련학과) 점수'] + applicant_data['관련 경험 점수'] +
                          applicant_data['취창업 의지 점수'] + applicant_data['수업 참여 의지 점수'])

# Save the updated applicant list with scores to an Excel file
updated_applicant_list_path = "/mnt/data/Updated_Applicant_List_with_Scores.xlsx"
applicant_data.to_excel(updated_applicant_list_path, index=False)
